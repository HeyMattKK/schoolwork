import java.awt.Canvas;

class FlagCanvas extends Canvas
{
   private int flagWidth;
   private int flagHeight;
   
   public FlagCanvas(int width)
   {
      flagWidth = width;
      flagHeight = width*2/3;
   }
   
   @Override
   public void paint(Graphics g)
   {
      // Draw the red rectangle background of the flag
      g.setColor(Color.RED);
      g.fillRect(0, 0, flagWidth, flagHeight);
      double ratio = flagWidth/30.0;
      
      /*
               5,5, 3,-pi/2
               10,2,1,pi-atan(3/5))
               12,4,1,pi-atan(1/7))
               12,7,1,pi + atan(2/7))
               10,9,1,pi+atan(4/5))
              */
      paintStar(ratio*5, ratio*5, ratio*3, 3*Math.PI/2, g);
      paintStar(ratio*10, ratio*2, ratio, Math.PI-(Math.atan( 0.6)), g);
      paintStar(ratio*12, ratio*4, ratio, Math.PI-Math.atan(1/7.0), g);
      paintStar(ratio*12, ratio*7, ratio, Math.PI + Math.atan(2.0/7), g);
      paintStar(ratio*10, ratio*9, ratio, Math.PI+Math.atan(0.8), g);
      
   }
   int [][] vertices(double cx, double cy, double r, double angle)
   {
      double outerRadius = 1;
      double innerRadius = outerRadius * Math.sin(Math.PI / 10) / Math.sin(3*Math.PI / 10); // Calculate inner radius based on trigonometry
      
      double angleIncrement = ( Math.PI / 5);
      double startAngle = angle;
      int [][] d = new int[2][10];
      d[0] = new int[10]; d[1] = new int[10];
      for (int i = 0; i < 10; i++)
      {
         angle = startAngle + i * angleIncrement;
         double radius = r*((i%2)==0? outerRadius:innerRadius);
         double x = radius * Math.cos(angle) + cx;
         double y = radius * Math.sin(angle) + cy;
         d[0][i] = (int)Math.round(x);
         d[1][i] = (int)Math.round(y);
         System.out.println("Outer vertex: ("+ x + ", "+ y + ")");
      }
      return d;
   }
   // Method to paint a five-pointed star
   public void paintStar(double centerX, double centerY, double radius,double ang, Graphics g)
   {
      g.setColor(Color.YELLOW);
      int [][] d = vertices(centerX,centerY,radius,ang);
      g.fillPolygon(d[0], d[1],10);
      
   }
   
   public static void main(String[] args)
   {
      JFrame frame = new JFrame("Chinese National Flag Simulation");
      frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
      FlagCanvas canvas = new FlagCanvas(1152);
      frame.setSize(canvas.flagWidth, canvas.flagHeight);
      frame.getContentPane().add(canvas);
      frame.setVisible(true);
   }
}

import java.util.*;

public class Sum
{
   public static void main(String[] args)
   {
      double sum = 0.0;
      Scanner s = new Scanner(System.in);
      
      // Skip the header line (the first line with "Name: Amount")
      if (s.hasNextLine())
      {
         s.nextLine(); // Skip the first line
         
      }
      
      // Read and process each line
      while (true)
      {
         try
         {
            String line = s.nextLine(); // Read the next line
            if (line.equals("The end"))
            {
               // Stop if "The end" is found
               break;
            }
            
            // Split the line by colon and get the second part (amount)
            String[] parts = line.split(":");
            if (parts.length >= 2)
            {
               // Trim to remove any spaces around the amount and parse the number
               String amountStr = parts[1].trim();
               double amount = Double.parseDouble(amountStr);
               sum += amount; // Add the amount to the sum
               
            }
         }
         catch (Exception e)
         {
            break; // If there's an exception, stop processing
            
         }
      }
      
      System.out.println("sum="+ sum); // Output the final sum
      s.close(); // Close the scanner
      
   }
}

public class EditableDatasheet extends JFrame
{
   private static final int ROWS = 5; // 5
   private static final int COLS = 3; // 3
   private String[][] data = new String[ROWS][COLS];
   private String[] columnNames =
   {
      "ID", "Grade", "Score"

   public EditableDatasheet(String filename)
   {
      // 
      setTitle("Editable Datasheet");
      setSize(800, 600);
      setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
      
      // 
      readDataFromFile(filename);

            saveData(filename);

      }
      );
      fileMenu.add(saveItem);
      menuBar.add(fileMenu);
      setJMenuBar(menuBar);

      JPanel panel = new JPanel();

      for (int col = 0; col < COLS; col++)
      {
         JLabel lbl = new JLabel(columnNames[col]);
         lbl.setName("num0_"+ col); // 
         panel.add(lbl);
      }
      
      // 
      for (int row = 1; row < ROWS + 1; row++)
      {
         for (int col = 0; col < COLS; col++)
         {
            if (col == 1)
            {
               //  (Grade)  JComboBox
               JComboBox<String> cmb = new JComboBox<>(new String[]
               {
                  "A", "B", "C", "D", "F"
               }
               );
               cmb.setName("num"+ row + "_"+ col);
               cmb.setSelectedItem(data[row - 1][col]);
               panel.add(cmb);
            }
            else
            {
               //  JTextField
               JTextField txt = new JTextField(data[row - 1][col]);
               txt.setName("num"+ row + "_"+ col);
               panel.add(txt);
            }
         }
      }
      
      add(panel);
   }
   
   // 
   private void readDataFromFile(String filename)
   {
      try (BufferedReader br = new BufferedReader(new FileReader(filename)))
      {
         String line;
         int row = 0;
         while ((line = br.readLine()) != null && row < ROWS)
         {
            String[] tokens = line.split(",");
            for (int col = 0; col < COLS; col++)
            {
               data[row][col] = tokens[col];
            }
            row++;
         }
      }
      catch (IOException e)
      {
         e.printStackTrace();
      }
   }
   
   // 
   private void saveData(String filename)
   {
      try (BufferedWriter bw = new BufferedWriter(new FileWriter(filename)))
      {
         for (int row = 0; row < ROWS; row++)
         {
            for (int col = 0; col < COLS; col++)
            {
               Component comp = get(row + 1, col); // 
               String value = "";
               if (comp instanceof JTextField)
               {
                  value = ((JTextField) comp).getText();
               }
               else if (comp instanceof JComboBox)
               {
                  value = (String) ((JComboBox<?>) comp).getSelectedItem();
               }
               bw.write(value);
               if (col < COLS - 1)
               {
                  bw.write(",");
               }
            }
            bw.newLine();
         }
      }
      catch (IOException e)
      {
         e.printStackTrace();
      }
   }
   
   // 
   private Component get(int row, int col)
   {
      Component[] components = getContentPane().getComponents();
      for (Component comp : components)
      {
         if (comp.getName().equals("num"+ row + "_"+ col))
         {
            return comp;
         }
      }
      return null;
   }

   }

         try
         {
            File file = new File(path);
            Scanner scanner = new Scanner(file);
            scanner.useDelimiter("\n");
            StringBuilder content = new StringBuilder();
            while (scanner.hasNext())
            {
               content.append(scanner.next()).append("\n");

         }
      }
   }
